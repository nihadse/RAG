Just edit the FILE paths below, then run: python qwen_chatbot.py

import subprocess, sys

# ── STEP 1: Install dependencies ──────────────────────────────────────────────
print("⏳ Installing dependencies...")
subprocess.check_call([sys.executable, "-m", "pip", "install", "-q",
    "transformers", "accelerate", "pdfplumber", "python-docx", "torch"])
print("✅ Dependencies ready.\n")

# ── STEP 2: Set your file paths ───────────────────────────────────────────────
FILE_1 = "your_file_1.pdf"     # ← CHANGE THIS
FILE_2 = "your_file_2.docx"    # ← CHANGE THIS (or set to None)
FILE_3 = None                   # ← CHANGE THIS (or keep None)

MODEL_NAME = "Qwen/Qwen2.5-1.5B-Instruct"   # ← change to 3B or 7B if you have more RAM

# ─────────────────────────────────────────────────────────────────────────────

from pathlib import Path

# ── STEP 3: Extract text from files ───────────────────────────────────────────
def extract_pdf(path):
    import pdfplumber
    parts = []
    with pdfplumber.open(path) as pdf:
        for i, page in enumerate(pdf.pages):
            t = page.extract_text()
            if t:
                parts.append(f"[Page {i+1}]\n{t}")
    return "\n\n".join(parts)

def extract_docx(path):
    from docx import Document
    doc = Document(path)
    parts = [p.text.strip() for p in doc.paragraphs if p.text.strip()]
    for table in doc.tables:
        for row in table.rows:
            row_text = " | ".join(c.text.strip() for c in row.cells if c.text.strip())
            if row_text:
                parts.append(row_text)
    return "\n".join(parts)

def extract(path):
    if path is None:
        return None
    p = Path(path)
    if not p.exists():
        print(f"⚠️  File not found: {path}")
        return None
    ext = p.suffix.lower()
    if ext == ".pdf":
        return extract_pdf(path)
    elif ext in (".docx", ".doc"):
        return extract_docx(path)
    else:
        print(f"⚠️  Unsupported format: {ext}")
        return None

documents = {}
for label, path in [("File 1", FILE_1), ("File 2", FILE_2), ("File 3", FILE_3)]:
    text = extract(path)
    if text:
        name = Path(path).name
        documents[name] = text
        print(f"✅ {label} — {name} — {len(text.split()):,} words extracted")

if not documents:
    print("❌ No documents loaded. Edit the FILE paths at the top of this script.")
    sys.exit(1)

print(f"\n📚 {len(documents)} document(s) ready.\n")

# ── STEP 4: Load Qwen model ───────────────────────────────────────────────────
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

print(f"⏳ Loading {MODEL_NAME} ...")
print("   (First run downloads from HuggingFace — may take a few minutes)\n")

tokenizer = AutoTokenizer.from_pretrained(MODEL_NAME, trust_remote_code=True)

device = "cuda" if torch.cuda.is_available() else "cpu"
dtype  = torch.float16 if device == "cuda" else torch.float32

model = AutoModelForCausalLM.from_pretrained(
    MODEL_NAME,
    torch_dtype=dtype,
    device_map="auto" if device == "cuda" else None,
    trust_remote_code=True,
)
if device == "cpu":
    model = model.to("cpu")
model.eval()

print(f"✅ Model loaded on: {device.upper()}")
if device == "cpu":
    print("⚠️  CPU mode — responses will be slow.\n")

# ── STEP 5: Build system prompt ───────────────────────────────────────────────
CHARS_PER_DOC = 6000

def build_system_prompt(docs, max_chars=CHARS_PER_DOC):
    sections = []
    for name, text in docs.items():
        truncated = text[:max_chars] + "\n...[truncated]" if len(text) > max_chars else text
        sections.append(f"=== {name} ===\n{truncated}")
    context = "\n\n".join(sections)
    return (
        "You are a helpful document assistant. "
        "Answer questions strictly based on the documents provided. "
        "If the answer is not in the documents, say so clearly.\n\n"
        f"DOCUMENTS:\n{context}"
    )

system_prompt = build_system_prompt(documents)
chat_history  = []

# ── STEP 6: Chat loop ─────────────────────────────────────────────────────────
def ask(question, max_new_tokens=512, temperature=0.3):
    global chat_history

    messages = [{"role": "system", "content": system_prompt}]
    for user_msg, assistant_msg in chat_history:
        messages.append({"role": "user",      "content": user_msg})
        messages.append({"role": "assistant", "content": assistant_msg})
    messages.append({"role": "user", "content": question})

    text = tokenizer.apply_chat_template(
        messages, tokenize=False, add_generation_prompt=True
    )
    inputs = tokenizer([text], return_tensors="pt").to(model.device)

    with torch.no_grad():
        output_ids = model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            temperature=temperature,
            do_sample=temperature > 0,
            pad_token_id=tokenizer.eos_token_id,
        )

    new_ids  = output_ids[0][inputs["input_ids"].shape[1]:]
    response = tokenizer.decode(new_ids, skip_special_tokens=True).strip()
    chat_history.append((question, response))
    return response


print("\n" + "="*60)
print("💬 Document Chatbot ready! Type your question and press Enter.")
print("   Type 'reset' to clear history, 'quit' to exit.")
print("="*60 + "\n")

while True:
    try:
        question = input("👤 You: ").strip()
    except (EOFError, KeyboardInterrupt):
        print("\n👋 Bye!")
        break

    if not question:
        continue
    if question.lower() == "quit":
        print("👋 Bye!")
        break
    if question.lower() == "reset":
        chat_history = []
        print("🔄 History cleared.\n")
        continue

    print("🤖 Qwen: ", end="", flush=True)
    answer = ask(question)
    print(answer)
    print()
