def create_pdf_from_text(text):
    pdf = FPDF()
    pdf.add_page()
    pdf.set_font("Arial", size=12)

    for line in text.split("\n"):
        pdf.multi_cell(0, 10, line)

    # Generate PDF as string
    pdf_output = pdf.output(dest='S').encode('latin-1')  # PDF raw bytes
    return pdf_output
