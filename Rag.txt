    SUM(AUTO_CLOSE_COUNTER) AS AUTO_CLOSE_COUNTER,
    SUM(TT_COUNTER) AS TT_COUNTER,

    CASE
        WHEN SUM(TT_COUNTER) < 1 THEN 1
        ELSE FLOOR(
            SUM(AUTO_CLOSE_COUNTER) / SUM(TT_COUNTER) * 10000
        ) / 10000
    END AS AUTO_CLOSE

FROM
    $(if(
        timeType = 'Monthly',
        'RPT_INNER.T_OOREDOO_TT_MONTH',
        'RPT_INNER.T_OOREDOO_TT_DAY'
    ))

WHERE
    ${if(timeType='Monthly', 'CLOSE_MONTH', 'CLOSE_DAY')} >=
    '$(switch(
        inarray(timeType, ["Monthly","Weekly"]),
        1, if(len(startTimeMonth) > 0,
              CONCATENATE(startTimeMonth, "-01"),
              ""),
        2, FORMAT(
              DATEDELTA(startTimeWeek, weekday(startTimeWeek)),
              "yyyy-MM-dd"
           ),
        0, FORMAT(startTime, "yyyy-MM-dd")
    ))'

AND
    ${if(timeType='Monthly', 'CLOSE_MONTH', 'CLOSE_DAY')} <=
    '$(switch(
        inarray(timeType, ["Monthly","Weekly"]),
        1, FORMAT(
              DATEDELTA(
                  MONTHDELTA(
                      FORMAT(CONCATENATE(endTimeMonth, "-01"), "yyyy-MM-01"),
                      1
                  ),
                  -1
              ),
              "yyyy-MM-dd 23:59:59"
           ),
        2, FORMAT(
              DATEDELTA(
                  endTimeWeek,
                  7 - weekday(endTimeWeek) - 1
              ),
              "yyyy-MM-dd 23:59:59"
           ),
        0, FORMAT(endTime, "yyyy-MM-dd 23:59:59")
    ))'

$(if(
    len(outageTT_alarm) = 0 ||
    outageTT_alarm = "" ||
    outageTT_alarm = "all",
    "",
    " AND UPPER(ALARMNAME) IN (" + outageTT_alarm + ") "
))

$(if(
    len(outageTT_sitedownflag) = 0 ||
    outageTT_sitedownflag = "" ||

    "",
    " AND UPPER(SITEDOWNFLAG) IN (" + outageTT_sitedownflag + ") "
))

$(if(

))

        ELSE FLOOR(SUM(AUTO_TT_COUNTER) / SUM(TT_COUNTER) * 10000) / 10000
    END AS AUTO_TT
FROM RPT_INNER.T_OOREDOO_TT_DAY
WHERE CREATE_DAY >= TO_DATE('2026-05-01', 'YYYY-MM-DD')
  AND CREATE_DAY <= TO_DATE('2026-05-31 23:59:59', 'YYYY-MM-DD HH24:MI:SS')
GROUP BY TO_CHAR(CREATE_DAY, 'YYYY-MM-DD')
ORDER BY T;
